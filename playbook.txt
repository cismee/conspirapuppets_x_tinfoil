Scenario 2: LP Creation Fails Due to Gas
bash# Use the retry script
forge script script/RetryLP.s.sol \
  --rpc-url $BASE_RPC_URL \
  --broadcast \
  --private-key $PRIVATE_KEY

# If that fails, use emergency recovery diagnostic
forge script script/EmergencyRecover.s.sol \
  --rpc-url $BASE_RPC_URL
Scenario 3: Sell-out Race Condition
bash# Check if completion was triggered
cast call $CONSPIRAPUPPETS_ADDRESS \
  "mintCompleted()(bool)" \
  --rpc-url $BASE_RPC_URL

# If false despite sell-out, manually trigger
cast send $CONSPIRAPUPPETS_ADDRESS \
  "completeMint()" \
  --private-key $PRIVATE_KEY \
  --rpc-url $BASE_RPC_URL
Scenario 4: Timing Edge Case (L2 Timestamp)
bash# Check timing status
cast call $CONSPIRAPUPPETS_ADDRESS \
  "isTimingEdgeCase()(bool,string)" \
  --rpc-url $BASE_RPC_URL

# If stuck, force reschedule
cast send $CONSPIRAPUPPETS_ADDRESS \
  "forceScheduleLPCreation()" \
  --private-key $PRIVATE_KEY \
  --rpc-url $BASE_RPC_URL
